\definecolor{ipython_frame}{RGB}{207, 207, 207}

\definecolor{ipython_red}{RGB}{186, 33, 33}

	basicstyle=\tiny\ttfamily,
    numberstyle={\tiny \color{mygrey}},% size of the numbers

    emph=[1]{self}, emphstyle=[1]\color{ipython_light_magenta},

	% Comments and Docstrings:
    commentstyle=\color{ipython_grey}\tiny,
    stringstyle=\color{ipython_green},

    keepspaces=true,
    showstringspaces=false,
    rulecolor=\color{ipython_frame},
    frame=single,
    frameround={t}{t}{t}{t},

    keywordstyle=\color{ipython_dark_blue},
